cmake_minimum_required(VERSION 3.16)
set(CMAKE_CXX_STANDARD 20)
project(app)


file(GLOB SOURCES "sources/*.cpp" "sources/*.h" "vendor/base_64/base64.h" "vendor/base_64/base64.cpp")
source_group("Source Files" FILES ${SOURCES})

# Создание группы Login
file(GLOB LOGIN_FILES "sources/LoginWidget/*.cpp" "sources/LoginWidget/*.h")
source_group("Source Files\\LoginUI Files" FILES ${LOGIN_FILES})

file(GLOB PROFILE_EDITOR_FILES "sources/ProfileEditorWidget/*.cpp" "sources/ProfileEditorWidget/*.h")
source_group("Source Files\\ProfileEditor Files" FILES ${PROFILE_EDITOR_FILES})

file(GLOB CHATS_FILES "sources/ChatsWidget/*.cpp" "sources/ChatsWidget/*.h")
source_group("Source Files\\ChatsUI Files" FILES ${CHATS_FILES})

file(GLOB GREET_FILES "sources/GreetWidget/*.cpp" "sources/GreetWidget/*.h")
source_group("Source Files\\GreetUI Files" FILES ${GREET_FILES})

# Создание группы Network
file(GLOB NETWORK_FILES "sources/Network/*.cpp" "sources/Network/*.h")
source_group("Source Files\\Network Files" FILES ${NETWORK_FILES})

set(CMAKE_PREFIX_PATH "C:/Qt/6.5.1")
find_package(Qt6 COMPONENTS Widgets REQUIRED)
qt_standard_project_setup()
set(CMAKE_AUTOMOC ON)

include_directories("sources" "sources/LoginWidget" "sources/ChatsWidget" "sources/Network" "sources/GreetWidget" "sources/ProfileEditorWidget" "vendor" "vendor/base_64")
link_directories("vendor/sqlite")

qt6_add_resources(QRC_FILES resources/LoginWidget/loginResources.qrc resources/ChatsWidget/chatsResources.qrc resources/GreetWidget/greetResources.qrc)


add_executable(${PROJECT_NAME} ${SOURCES} ${LOGIN_FILES} ${CHATS_FILES} ${NETWORK_FILES} ${QRC_FILES} ${GREET_FILES} ${PROFILE_EDITOR_FILES})
target_link_libraries(${PROJECT_NAME} Qt6::Widgets ws2_32 sqlite.lib)
